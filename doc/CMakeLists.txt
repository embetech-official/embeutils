include(FindDoxygen)

if (NOT DOXYGEN_FOUND)
  message(STATUS "Doxygen not found, documentation will not be generated")
  return()
endif ()

set(DOXYGEN_PROJECT_NAME "EMBENET Utilities")
set(USE_MDFILE_AS_MAINPAGE ${CMAKE_CURRENT_SOURCE_DIR}/embeutils.md)
set(DOXYGEN_LAYOUT_FILE ${CMAKE_CURRENT_SOURCE_DIR}/DoxygenLayout.xml)
set(DOXYGEN_EXAMPLE_PATH ${PROJECT_SOURCE_DIR}/examples)
set(DOXYGEN_IMAGE_PATH ${CMAKE_CURRENT_SOURCE_DIR})

set(DOXYGEN_OUTPUT_DIRECTORY ${CMAKE_INSTALL_PREFIX}/doc)
set(DOXYGEN_GENERATE_TAGFILE ${DOXYGEN_OUTPUT_DIRECTORY}/html/embeutils.tag)

set(DOXYGEN_WARN_AS_ERROR YES)

set(DOXYGEN_OPTIMIZE_OUTPUT_FOR_C YES)
set(DOXYGEN_TIMESTAMP YES)
set(DOXYGEN_CASE_SENSE_NAMES NO)
set(DOXYGEN_SHOW_INCLUDE_FILES NO)
set(DOXYGEN_SORT_MEMBER_DOCS NO)
set(DOXYGEN_ENABLE_PREPROCESSING YES)
set(DOXYGEN_MACRO_EXPANSION YES)
set(DOXYGEN_EXPAND_ONLY_PREDEF YES)
set(DOXYGEN_PREDEFINED "_DOXYGEN_" "__attribute__((x))=")
set(DOXYGEN_JAVADOC_AUTOBRIEF YES)
set(DOXYGEN_REFERENCES_LINK_SOURCE NO)
set(DOXYGEN_VERBATIM_HEADERS NO)
set(DOXYGEN_HTML_COLORSTYLE DARK)
set(DOXYGEN_GENERATE_LATEX NO)
set(DOXYGEN_ALIASES
    [[purpose=@par Purpose:]] [[license=@par License:]] [[compiler=@par Compiler:]]
    [[ hardware=@par Hardware:]] [[comments=@par Comments:]] [[doc=@par Documented using:]]
)
set(DOXYGEN_EXTRACT_ALL YES)
set(DOXYGEN_HTML_COLORSTYLE AUTO_LIGHT)
set(DOXYGEN_GENERATE_TREEVIEW YES)
set(DOXYGEN_SEARCHENGINE YES)
set(DOXYGEN_PREDEFINED "__attribute__((x))=")
set(DOXYGEN_TYPEDEF_HIDES_STRUCT YES)
set(DOXYGEN_ENUM_VALUES_PER_LINE 1)
set(WARN_IF_UNDOCUMENTED YES)
set(WARN_IF_DOC_ERROR YES)
set(WARN_IF_INCOMPLETE_DOC YES)
set(WARN_NO_PARAMDOC YES)

doxygen_add_docs(
  embeutils_doc ${CMAKE_CURRENT_SOURCE_DIR} ${PROJECT_SOURCE_DIR}/src/include/embetech
  COMMENT "Generate documentation"
)
